// NatureAuth Firebase Entegrasyonu const NATURE_AUTH_DB = 'https://PROJE.firebasedatabase.app'; async function ncLogin(username, password, server) { const hash = await sha256(password + username); const res = await fetch(NATURE_AUTH_DB + '/users/' + server + '_' + username + '.json'); const user = await res.json(); if (!user) return { ok: false, err: 'Kullanıcı bulunamadı.' }; if (user.banned) return { ok: false, err: 'Hesap yasaklandı.' }; if (user.passwordHash !== hash) return { ok: false, err: 'Şifre yanlış.' }; const token = generateToken(); await fetch(NATURE_AUTH_DB + '/sessions/' + token + '.json', { method: 'PUT', body: JSON.stringify({ username, server, isAdmin: user.isAdmin, ts: Date.now() }) }); return { ok: true, token, username, isAdmin: user.isAdmin }; }
